Part K: Quasi-Static Growth Equation Solver
Here is a clean implementation of the modified growth equation using the quasi-static $  G_{\rm eff}  $.
Python Version (Recommended)
PythonCopyimport numpy as np
from scipy.integrate import solve_ivp

def growth_equation_quasistatic(a, y, cosmo_params, prtoe_params):
    """
    Quasi-static growth equation:
    D'' + (2 + H'/H) D' - (3/2) Omega_m(a) * (G_eff/G) * D = 0
    """
    D, dD_da = y
    
    a = max(a, 1e-6)
    H = cosmo_params['H0'] * np.sqrt(cosmo_params['Omega_m']/a**3 + cosmo_params['Omega_L'])
    
    # Get background quantities (you would pull these from CLASS or your background module)
    F = prtoe_params['F']
    F_phi = prtoe_params['F_phi']
    m_eff2 = prtoe_params['m_eff2']
    
    k = prtoe_params.get('k', 0.1)  # h/Mpc
    
    # Compute G_eff
    k2_over_a2 = (k / a)**2
    denom = k2_over_a2 + m_eff2
    Geff_over_G = (1.0 / F) * (1.0 + 2.0 * (F_phi / F)**2 / denom)
    
    Omega_m_a = cosmo_params['Omega_m'] / (a**3 * (cosmo_params['Omega_m']/a**3 + cosmo_params['Omega_L']))
    
    # Growth equation
    d2D_da2 = - (2 + cosmo_params.get('H_prime_over_H', 0)) * dD_da \
              + (3.0/2.0) * Omega_m_a * Geff_over_G * D
    
    return [dD_da, d2D_da2]


def solve_growth_quasistatic(cosmo_params, prtoe_params, a_span=(1e-3, 1.0)):
    """Solve for the growth factor D(a) in quasi-static approximation."""
    y0 = [1e-5, 1e-5]  # Initial conditions at early time
    sol = solve_ivp(
        growth_equation_quasistatic,
        a_span,
        y0,
        args=(cosmo_params, prtoe_params),
        dense_output=True,
        rtol=1e-6
    )
    return sol.t, sol.y[0]

1. Full Quasi-Static Power Spectrum Modification
Here is a practical implementation to compute the modified linear matter power spectrum in the quasi-static approximation.
Python Function
PythonCopyimport numpy as np
from scipy.interpolate import interp1d

def compute_modified_power_spectrum(
    k_array,
    z,
    P_lcdm,                    # Base LambdaCDM power spectrum P(k) at redshift z
    cosmo_params,
    prtoe_params,
    growth_lcdm=None         # Optional: precomputed LCDM growth factor
):
    """
    Compute modified matter power spectrum in quasi-static approximation.
    
    P_PRTOE(k, z) ≈ P_ΛCDM(k, z) * (D_PRTOE(k, z) / D_ΛCDM(z))^2 * (G_eff(k, z) / G)^2
    """
    a = 1.0 / (1.0 + z)
    
    # Solve growth in quasi-static limit
    a_vals, D_prtoe = solve_growth_quasistatic(cosmo_params, prtoe_params)
    
    # Growth factor at desired redshift
    D_prtoe_z = np.interp(a, a_vals, D_prtoe)
    
    if growth_lcdm is None:
        # Simple approximation if not provided
        D_lcdm_z = a  # normalized at early times
    else:
        D_lcdm_z = np.interp(a, growth_lcdm['a'], growth_lcdm['D'])
    
    growth_ratio = (D_prtoe_z / D_lcdm_z)**2
    
    # Compute G_eff at each k
    Geff_array = np.zeros_like(k_array)
    for i, k in enumerate(k_array):
        result = compute_prtoe_quasistatic(
            k=k,
            a=a,
            F=prtoe_params['F'],
            F_phi=prtoe_params['F_phi'],
            m_eff2=prtoe_params['m_eff2'],
            H=cosmo_params['H0'] * np.sqrt(cosmo_params['Omega_m']/a**3 + cosmo_params['Omega_L']),
            Omega_m=cosmo_params['Omega_m']
        )
        Geff_array[i] = result['Geff_over_G']
    
    # Modified power spectrum
    P_modified = P_lcdm * growth_ratio * (Geff_array)**2
    
    return P_modified, Geff_array, growth_ratio

    double Geff_over_G = (1.0 / F) * (1.0 + 2.0 * pow(F_phi / F, 2) / denom);
    double eta         = 1.0 + 2.0 * pow(F_phi / F, 2) / denom;

    /* === Modified Poisson constraint (0,0) === */
    double rho_m_delta = ppw->delta_m;   // or compute from matter species
    double source_term = F_phi * delta_phi * (3.0 * (H*H + H_prime/a));

    double Phi_prime = -H * Phi 
                       - (rho_m_delta + source_term) / (2.0 * F * k2_over_a2);

    /* === Slip relation: get Psi from Phi and delta_phi === */
    double Psi = Phi - (F_phi / F) * delta_phi;

    /* === Perturbed Klein-Gordon (quasi-static simplified version) === */
    double ddelta_phi_prime = 
        - (3.0*H + F_phi * pvecback[pba->index_bg_dphi_prtoe]/F / a) * ddelta_phi
        - (k2_over_a2 + m_eff2) * delta_phi
        + (F_phi / F) * 3.0 * (H_prime/a + 2.0*H*H) * (3.0*Psi + 2.0*Phi);

    /* === Store metric potentials for source terms === */
    ppw->pvecmetric[ppw->index_mt_Phi_prtoe] = Phi;
    ppw->pvecmetric[ppw->index_mt_Psi_prtoe] = Psi;
    ppw->pvecmetric[ppw->index_mt_Geff_prtoe] = Geff_over_G;   // useful for sources

    /* === Write derivatives === */
    dy[ppw->pv->index_pt_delta_prtoe]  = ddelta_phi;
    dy[ppw->pv->index_pt_ddelta_prtoe] = ddelta_phi_prime;
    dy[ppw->pv->index_pt_Phi_prtoe]    = Phi_prime;
}

Phi_prime and Psi_prime should be computed or stored earlier in the perturbation evolution.
